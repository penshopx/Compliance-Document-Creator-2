    let generatedDoc = "";
    let hasGenerated = false;
    document.getElementById('generatorForm').addEventListener('submit', async function(e) {
      e.preventDefault();
      if (hasGenerated) {
        document.getElementById('generatorStatus').textContent = currentLang === 'id'
          ? "Anda hanya dapat generate satu dokumen perusahaan."
          : "You can only generate one company document.";
        document.getElementById('generatorStatus').className = "text-red-600 mt-2";
        return;
      }
      const companyName = document.getElementById('companyName').value.trim();
      const companyDesc = document.getElementById('companyDesc').value.trim();
      const openaiKey = document.getElementById('openaiKey').value.trim();
      if (!companyName || !companyDesc || !openaiKey) return;
      document.getElementById('generatorStatus').textContent = currentLang === 'id'
        ? "Sedang memproses dokumen..."
        : "Generating document...";
      document.getElementById('generatorStatus').className = "text-primary-600 mt-2";
      document.getElementById('generateBtn').disabled = true;
      // Simulasi loading
      await new Promise(res => setTimeout(res, 1200));
      // Generate dokumen via OpenAI API
      try {
        // Prompt
        const prompt = currentLang === 'id'
          ? `Buatkan dokumen kebijakan anti penyuapan untuk perusahaan berikut:\nNama: ${companyName}\nDeskripsi: ${companyDesc}\nFormat: Bahasa Indonesia, struktur formal, minimal 5 poin kebijakan.`
          : `Create an anti-bribery policy document for the following company:\nName: ${companyName}\nDescription: ${companyDesc}\nFormat: English, formal structure, at least 5 policy points.`;
        // Fetch to OpenAI API
        const response = await fetch("https://api.openai.com/v1/chat/completions", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "Authorization": "Bearer " + openaiKey
          },
          body: JSON.stringify({
            model: "gpt-3.5-turbo",
            messages: [{role: "user", content: prompt}],
            max_tokens: 800,
            temperature: 0.2
          })
        });
        if (!response.ok) {
          throw new Error("Gagal menghubungi API OpenAI.");
        }
        const data = await response.json();
        const content = data.choices?.[0]?.message?.content || "";
        generatedDoc = content;
        document.getElementById('docPreview').value = generatedDoc;
        document.getElementById('downloadBtn').disabled = false;
        document.getElementById('resetBtn').disabled = false;
        document.getElementById('generatorStatus').textContent = currentLang === 'id'
          ? "Dokumen berhasil digenerate. Silakan edit atau download."
          : "Document generated successfully. You can edit or download.";
        document.getElementById('generatorStatus').className = "text-green-600 mt-2";
        hasGenerated = true;
        // Save to localStorage for monitoring
        localStorage.setItem('sgd_doc', JSON.stringify({
          companyName, companyDesc, content: generatedDoc, date: new Date().toISOString()
        }));
        updateMonitoringTable();
      } catch (err) {
        document.getElementById('generatorStatus').textContent = currentLang === 'id'
          ? "Gagal generate dokumen. Pastikan API Key benar."
          : "Failed to generate document. Please check your API Key.";
        document.getElementById('generatorStatus').className = "text-red-600 mt-2";
      }
      document.getElementById('generateBtn').disabled = false;
    });

    // =========================
    // Monitoring Section
    // =========================
    function getMonitoringData() {
      // Only one doc per user
      const doc = JSON.parse(localStorage.getItem('sgd_doc') || '{}');
      if (!doc.companyName) return [];
      return [[
        doc.companyName,
        doc.companyDesc,
        doc.date ? new Date(doc.date).toLocaleDateString() : "-",
        doc.assessment?.score || "-",
        doc.assessment?.note || "-"
      ]];
    }
    let monitoringGrid;
    function updateMonitoringTable(filter = "") {
      const data = getMonitoringData().filter(row =>
        !filter ||
        row[0].toLowerCase().includes(filter) ||
        row[1].toLowerCase().includes(filter)
      );
      if (monitoringGrid) monitoringGrid.updateConfig({data}).forceRender();
      else {
        monitoringGrid = new gridjs.Grid({
          columns: [
            {name: currentLang === 'id' ? "Nama Perusahaan" : "Company Name"},
            {name: currentLang === 'id' ? "Deskripsi" : "Description"},
            {name: currentLang === 'id' ? "Tanggal" : "Date"},
            {
              name: currentLang === 'id' ? "Nilai Assessment" : "Assessment Score",
              formatter: cell => gridjs.h('div', {
                className: cell >= 80 ? 'text-green-600 font-medium' : cell >= 60 ? 'text-yellow-600 font-medium' : 'text-red-600 font-medium',
                style: {whiteSpace: 'nowrap'}
              }, cell + (cell !== "-" ? "%" : ""))
            },
            {name: currentLang === 'id' ? "Catatan" : "Notes"}
          ],
          data,
          search: false,
          sort: true,
          pagination: false,
          language: {
            'noRecordsFound': currentLang === 'id' ? "Tidak ada data monitoring." : "No monitoring data."
          }
        }).render(document.getElementById('monitoringTable'));
      }
    }
    updateMonitoringTable();
    document.getElementById('monitoringSearch').addEventListener('keyup', function() {
      updateMonitoringTable(this.value.toLowerCase());
    });

    document.getElementById('exportMonitoring').addEventListener('click', function() {

      const a = document.createElement('a');

      a.download = "Monitoring-SMAP.csv";
      document.body.appendChild(a);

        document.body.removeChild(a);

      }, 100);
