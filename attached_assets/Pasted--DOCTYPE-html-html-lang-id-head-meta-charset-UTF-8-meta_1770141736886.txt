    // Render form fields for clause
    function renderClauseForm(key, customTitle) {
      const clauseForm = document.getElementById('clauseForm');
      const dynamicFieldsDiv = clauseForm;
      let fields = getFieldDefForKey(key);
      // If customTitle is set, prefill judulDokumen
      fields = fields.map(f => f.name === "judulDokumen" ? {...f, defaultValue: customTitle} : f);
      dynamicFieldsDiv.innerHTML = '';
      fields.forEach(field => {
        let label = field.label[currentLang] || field.label.id;
        let required = field.required ? 'required' : '';
        let id = field.name;
        let placeholder = field.placeholder ? (field.placeholder[currentLang] || field.placeholder.id) : '';
        let value = field.defaultValue || '';
        let fieldHtml = '';
        if (field.type === "text" || field.type === "date") {
          fieldHtml = `
            <div class="mb-4">
              <label for="${id}" class="block text-sm font-medium text-gray-700 mb-1">${label}${field.required ? ' *' : ''}</label>
              <input type="${field.type}" id="${id}" name="${id}" class="w-full px-4 py-2 border border-gray-300 rounded-md focus:ring-primary-400 focus:border-primary-400 transition" placeholder="${placeholder}" value="${value || ''}" ${required}>
            </div>
          `;
        } else if (field.type === "textarea") {
          fieldHtml = `
            <div class="mb-4">
              <label for="${id}" class="block text-sm font-medium text-gray-700 mb-1">${label}${field.required ? ' *' : ''}</label>
              <textarea id="${id}" name="${id}" rows="2" class="w-full px-4 py-2 border border-gray-300 rounded-md focus:ring-primary-400 focus:border-primary-400 transition" placeholder="${placeholder}" ${required}>${value || ''}</textarea>
            </div>
          `;
        } else if (field.type === "select") {
          fieldHtml = `
            <div class="mb-4">
              <label for="${id}" class="block text-sm font-medium text-gray-700 mb-1">${label}${field.required ? ' *' : ''}</label>
              <select id="${id}" name="${id}" class="w-full px-4 py-2 border border-gray-300 rounded-md focus:ring-primary-400 focus:border-primary-400 transition" ${required}>
                ${field.options.map(opt => `<option value="${opt.value}" ${value === opt.value ? 'selected' : ''}>${opt.label[currentLang] || opt.label.id}</option>`).join('')}
              </select>
            </div>
          `;
        }
        dynamicFieldsDiv.innerHTML += fieldHtml;
      });

      // Generator function for preview
      function generateDummyDocument(vars, fields, customVars, customTitle) {
        let lang = vars.bahasa === 'en' ? 'en' : 'id';
        let customVarsSection = '';
        if (customVars.length > 0 && customVars.some(v => v.name && v.value)) {
          customVarsSection = `<ul class="list-disc pl-5 mb-2">` +
            customVars.filter(v => v.name && v.value).map(v =>
              `<li><b>${v.name}:</b> ${v.value}</li>`
            ).join('') +
            `</ul>`;
        }
        // Compose body based on fields
        let body = `<h2 class="text-xl font-bold mb-2">${vars.judulDokumen || customTitle || '[Judul Dokumen]'}</h2>`;
        fields.forEach(field => {
          if (field.name === "judulDokumen") return;
          let label = field.label ? (field.label[lang] || field.label.id) : field.name;
          let value = vars[field.name];
          if (value) {
            if (field.type === "textarea") {
              body += `<p class="mb-2"><b>${label}:</b><br>${value.replace(/\n/g, "<br>")}</p>`;
            } else if (field.type === "date") {
              body += `<p class="mb-2"><b>${label}:</b> ${value}</p>`;
            } else if (field.type === "select") {
              // skip language field
            } else {
              body += `<p class="mb-2"><b>${label}:</b> ${value}</p>`;
            }
          }
        });
        if (customVarsSection) {
          body += `<div class="mb-2"><b>${lang === 'en' ? 'Additional Variables' : 'Variabel Tambahan'}:</b>${customVarsSection}</div>`;
        }
        body += `<div class="mt-4 italic text-gray-500">${lang === 'en' ? 'This is a dummy preview. Please edit as needed before saving or exporting.' : 'Ini adalah preview dummy. Silakan edit sebelum menyimpan atau mengekspor.'}</div>`;
        return body;
      }

        });
