        // --- CORE SECURITY ---
        async function generateHash(msg) {
            const encoder = new TextEncoder();
            const data = encoder.encode(msg);
            const hashBuffer = await crypto.subtle.digest('SHA-256', data);
            return Array.from(new Uint8Array(hashBuffer)).map(b => b.toString(16).padStart(2, '0')).join('').substring(0, 18).toUpperCase();
        }

        function showAlert(m, t) {
            const a = document.getElementById('alert-area');
            a.className = `mb-8 p-6 border-l-[10px] rounded-3xl shadow-xl ${t==='success'?'bg-green-50 border-green-500 text-green-800':'bg-red-50 border-red-500 text-red-800'}`;

        function smartCopy(btn, id) {
            const el = document.getElementById(id);
            const selection = window.getSelection();
            const range = document.createRange();
            range.selectNodeContents(el);
            selection.removeAllRanges();
            selection.addRange(range);
            try {
                document.execCommand('copy');
                const old = btn.textContent;
                btn.textContent = "✅ TERSALIN"; btn.style.backgroundColor = "#16a34a";
                setTimeout(() => { btn.textContent = old; btn.style.backgroundColor = ""; selection.removeAllRanges(); }, 2000);
            } catch(e) {}
        }
