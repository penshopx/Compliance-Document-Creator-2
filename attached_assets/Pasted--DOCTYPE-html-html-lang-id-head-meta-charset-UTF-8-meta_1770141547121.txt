    function getTemplateKey(docName) {
      return `pancek_template_${docName.replace(/\s+/g, '_').toLowerCase()}`;
    }

    document.getElementById('btn-save-template').addEventListener('click', function() {
      const form = document.getElementById('dynamic-generator-form');
      if (!form.checkValidity()) {
        form.reportValidity();
        return;
      }
      const values = getFormValues(form, currentDocTemplate);
      // Save to localStorage
      const key = getTemplateKey(currentDocName);
      let arr = JSON.parse(localStorage.getItem(key) || '[]');
      arr.push(values);
      localStorage.setItem(key, JSON.stringify(arr));
      // Show notification
      const notif = document.getElementById('template-saved-notif');
      notif.classList.remove('hidden');
      notif.style.opacity = 1;
      setTimeout(() => {
        notif.style.transition = 'opacity 0.5s';
        notif.style.opacity = 0;
        setTimeout(() => notif.classList.add('hidden'), 500);
      }, 1500);
      renderSavedTemplates(currentDocName);
    });

    function renderSavedTemplates(docName) {
      const key = getTemplateKey(docName);
      const arr = JSON.parse(localStorage.getItem(key) || '[]');
      const container = document.getElementById('saved-templates-list');
      container.innerHTML = '';
      if (!arr.length) {
        container.innerHTML = '<span class="text-gray-400">Belum ada template tersimpan.</span>';
        return;
      }
      arr.forEach((tpl, idx) => {
        const btn = document.createElement('button');
        btn.className = "px-3 py-2 bg-primary-100 hover:bg-primary-200 text-primary-800 rounded-md font-medium transition-colors flex items-center gap-2";
        btn.innerHTML = `<i class="fas fa-file-alt"></i>Template ${idx + 1}`;
        btn.addEventListener('click', function() {
          // Load template to form
          const form = document.getElementById('dynamic-generator-form');
          currentDocTemplate.fields.forEach(field => {
            if (form.elements[field.name]) {
              form.elements[field.name].value = tpl[field.name] || '';
            }
          });
        });
        // Delete button
        const delBtn = document.createElement('button');
        delBtn.className = "ml-2 text-red-500 hover:text-red-700";
        delBtn.innerHTML = `<i class="fas fa-trash"></i>`;
        delBtn.title = "Hapus template";
        delBtn.addEventListener('click', function(e) {
          e.stopPropagation();
          arr.splice(idx, 1);
          localStorage.setItem(key, JSON.stringify(arr));
          renderSavedTemplates(docName);
        });
        const wrapper = document.createElement('div');
        wrapper.className = "flex items-center";
        wrapper.appendChild(btn);
        wrapper.appendChild(delBtn);
        container.appendChild(wrapper);
      });
    }

      btnDownloadExcel.disabled = false;
    });

    // =========================
    // Accordion logic for checklist (Panduan)
    // =========================
    const accordion = document.getElementById('checklist-accordion');
    if (accordion) {
      accordion.querySelectorAll('button[type="button"]').forEach((btn, idx) => {
        btn.addEventListener('click', function() {
          const expanded = this.getAttribute('aria-expanded') === 'true';
          // Collapse all
          accordion.querySelectorAll('button[type="button"]').forEach(b => {
            b.setAttribute('aria-expanded', 'false');
            b.querySelector('i.fas.fa-chevron-down').classList.remove('rotate-180');
          });
          accordion.querySelectorAll('div[role="region"]').forEach(panel => panel.classList.add('hidden'));
          // Expand this
          if (!expanded) {
            this.setAttribute('aria-expanded', 'true');
            const panelId = this.getAttribute('aria-controls');
            document.getElementById(panelId).classList.remove('hidden');
            // Animate chevron
            this.querySelector('i.fas.fa-chevron-down').classList.add('rotate-180');
          }
        });
        // Keyboard accessibility
        btn.addEventListener('keydown', function(e) {
          if (e.key === 'Enter' || e.key === ' ') {
            e.preventDefault();
            btn.click();
          }
        });
      });
      // Expand first by default
      accordion.querySelectorAll('button[type="button"]')[0].setAttribute('aria-expanded', 'true');
      accordion.querySelectorAll('div[role="region"]')[0].classList.remove('hidden');
    }

    // Set focus to main content on navigation for accessibility
    function setMainFocus() {
      mainContent.setAttribute('tabindex', '-1');
      mainContent.focus();
    }
    [navHome, navGenerator, navChecklist, navInventarisasi, navContact].forEach(btn => {
      btn.addEventListener('click', setMainFocus);
    });

    // On page load, show home
    showSection(sectionHome);

    // Accessibility: close modals with ESC
    // =========================
    document.addEventListener('keydown', function(e) {
      if (e.key === 'Escape') {
        document.getElementById('tooltip-modal').classList.add('hidden');
        rekapModal.classList.add('hidden');
      }
    });
